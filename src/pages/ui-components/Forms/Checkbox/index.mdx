# Checkbox

O componente de caixa de seleção (Checkbox) permite que o usuário selecione um ou mais itens de um determinado grupo.

### Importação

```js
import Checkbox from '@eduzz/houston-ui/Forms/Checkbox';
```

### Variações

<Playground>
  {() => {
    const [checked, setChecked] = React.useState(false);
    return (
      <>
        <Checkbox checked={checked} onChange={setChecked}>
          Simples
        </Checkbox>
        <Checkbox helperText='Exemplo de utilização do Checkbox com descrição' checked={checked} onChange={setChecked}>
          Com descrição
        </Checkbox>
        <Checkbox errorMessage='Mensagem de erro' checked={checked} onChange={setChecked}>
          Com erro
        </Checkbox>
        <Checkbox disabled helperText='Checkbox sem margem' checked={checked} onChange={setChecked}>
          Desativado
        </Checkbox>
      </>
    );
  }}
</Playground>

### Formulário

Caso utilize o componente de caixa de seleção sem estar dentro de um contexto de formulário,
é possível referenciá-lo através da propriedade `form`.

```tsx
import useForm from '@eduzz/houston-forms/useForm';
import Checkbox from '@eduzz/houston-ui/Forms/Checkbox';
import Form from '@eduzz/houston-ui/Forms/Form';

interface IModel {
  check: boolean;
}

const MyComponent = () => {
  const form = useForm<IModel>({
    onSubmit: () => {}
  });

  return (
    <Form context={form}>
      <Checkbox name='check' helperText='Checkbox fora de um FormContext'>
        Exemplo formulário
      </Checkbox>
    </Form>
  );
};
```

### Utilizando método onChange (sem hook useForm)

<Playground>
  {() => {
    const [value, setValue] = React.useState(false);
    const handleChange = checked => {
      setValue(checked);
      console.log('current checkbox value', checked);
    };
    return (
      <Checkbox helperText='Checkbox utilizando método de onChange' onChange={handleChange} checked={value}>
        Exemplo com método onChange
      </Checkbox>
    );
  }}
</Playground>

| prop         | tipo       | obrigatório | padrão  | descrição                                                                                                                           |
| ------------ | ---------- | ----------- | ------- | ----------------------------------------------------------------------------------------------------------------------------------- |
| name         | `string`   | `true`      | -       |                                                                                                                                     |
| helperText   | `string`   | `false`     | -       |                                                                                                                                     |
| errorMessage | `string`   | `false`     | -       |                                                                                                                                     |
| multiple     | `boolean`  | `false`     | `false` |                                                                                                                                     |
| , `normal`   | `false`    | `none`      |         |
| checked      | `boolean`  | `false`     | `false` | Caso a prop `form` esteja configurada, o componente será checado através dos valores do formulário e essa propriedade será ignorada |
| onChange     | `function` | `false`     | -       | Função chamada quando o valor do checkbox é alterado                                                                                |
| disabled     | `boolean`  | `false`     | -       |                                                                                                                                     |
